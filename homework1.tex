\begin{flushleft}
  \textbf{普通电力二极管(General purpose Diode)}
\end{flushleft}

制造商：ABB\footnote{\url{http://category.alldatasheetcn.com/index.jsp?sSearchword=DIODE&selMfg=434&cate=1}}

\textbf{5SDD54N4000}
主要参数（$T_j=150^{\circ}C$）：
\begin{itemize}
\item  额定电压$V_{RR}$（反向重复施加的最大峰值电压）：3600$V$
\item  额定电流$I_{FR}$：5200$A$
\item  最大允许的均方根正向电流$I_{Frms}$：8200$A$
\item  最大允许非重复浪涌电流$I_{FSM}$：85000$A$
\item  正向最大电压降落$V_F$：1.23$V$
\end{itemize}



\textbf{5SDD50N5500}
主要参数（$T_j=150^{\circ}C$:
\begin{itemize}
\item  额定电压$V_{RR}$（反向重复施加的最大峰值电压）：5000$V$
\item  额定电流$I_{FR}$：4700$A$
\item  最大允许的均方根正向电流$I_{Frms}$：8200$A$
\item  最大允许非重复浪涌电流$I_{FSM}$：73000$A$
\item  正向最大电压降落$V_F$：1.34$V$
\end{itemize}
按照ABB普通电力二极管的命名方式，5S代表半导体器件，DD指普通二极管，DF指快恢复二极管。中间两位数字越大，额定电流越大，最后四位数字代表额定电压。

\begin{flushleft}
  \textbf{快恢复二极管(High speed or fast recovery Diode)}
\end{flushleft}
制造商：Power Semiconductors(POSEICO)\footnote{\url{http://category.alldatasheetcn.com/index.jsp?sSearchword=DIODE&sPage=1&selMfg=122}}

\textbf{ARF673}
主要参数（$T_j=25^{\circ}C$）：
\begin{itemize}
\item  额定电压$V_{RR}$（反向重复施加的最大峰值电压）：4500$V$
\item  额定电流$I_{FR}$：990$A$
\item  反向电压有效值$V_{RMS}$：4600$V$
\item  最大允许非重复浪涌电流$I_{FSM}$：1500$A$
\item  正向最大电压降落$V_F$：1.7$V$
\item  恢复时间$t_rr$：1.9$\mu s$
\end{itemize}

\textbf{ARF672}
主要参数（$T_j=25^{\circ}C$）：
\begin{itemize}
\item  额定电压$V_{RR}$（反向重复施加的最大峰值电压）：4500$V$
\item  额定电流$I_{FR}$：935$A$
\item  反向电压有效值$V_{RMS}$：4600$V$
\item  最大允许非重复浪涌电流$I_{FSM}$：1500$A$
\item  正向最大电压降落$V_F$：1.8$V$
\item  恢复时间$t_rr$：1.9$\mu s$
\end{itemize}

按POSEICO的编号规则AR指的是普通电力二极管，ARF指的是快恢复二极管。

\begin{flushleft}
  \textbf{肖特基二极管(Schotty Diode)}
\end{flushleft}
制造商：Fairchild Semiconductor\footnote{\url{http://category.alldatasheet.com/index.jsp?sSearchword=SCHOTTY&selMfg=51}}

\textbf{MBR20200CT}
主要参数（$T_j=25^{\circ}C$）：
\begin{itemize}
\item  额定电压$V_{RR}$（反向重复施加的最大峰值电压）：200$V$
\item  额定电流$I_{FR}$：20$A$(per device)
\item  反向直流电压最大值$V_R$：200$V$
\item  最大允许非重复浪涌电流$I_{FSM}$：150$A$
\item  正向最大电压降落$V_F$：1.0$V$(20$A$)
%\item  恢复时间$t_rr$：1.9$\mu s$
\end{itemize}

\textbf{MBR20150CT}
主要参数（$T_j=25^{\circ}C$）：
\begin{itemize}
\item  额定电压$V_{RR}$（反向重复施加的最大峰值电压）：150$V$
\item  额定电流$I_{FR}$：20$A$(per device)
\item  反向直流电压最大值$V_R$：200$V$
\item  最大允许非重复浪涌电流$I_{FSM}$：150$A$
\item  正向最大电压降落$V_F$：0.95$V$(20$A$)
%\item  恢复时间$t_rr$：1.9$\mu s$
\end{itemize}
仙童的这两种二极管都是双二极管并联，最后三位数200/150指的是额定电压值，中间两位数20 指的是额定电流。

\textbf{5STP34H1601}
主要参数：
\begin{itemize}
\item  额定电压$V_{R}$：1600$V$
\item  额定电流$I_{T(AV)M}$：3370$A$
\item  通态电压降落$V_T$：1.2$V$
\item  最大允许非重复浪涌电流$I_{FSM}$：49000$A$
\item  维持电流$I_H$（$T_j=25^{\circ}C$）：170$mA$
\item  擎住电流$I_L$（$T_j=25^{\circ}C$）：1500$mA$
\item  开通时间$t_{on}$：2$\mu s$
\item  关断时间$t_{off}$：200$\mu s$
%\item  恢复时间$t_rr$：1.9$\mu s$
\end{itemize}


\textbf{5STP50Q1800}
主要参数：
\begin{itemize}
\item  额定电压$V_{R}$：1800$V$
\item  额定电流$I_{T(AV)M}$：6100$A$
\item  通态电压降落$V_T$：1.04$V$
\item  最大允许非重复浪涌电流$I_{FSM}$：94000$A$
\item  维持电流$I_H$（$T_j=25^{\circ}C$）：100$mA$
\item  擎住电流$I_L$（$T_j=25^{\circ}C$）：500$mA$
\item  开通时间$t_{on}$：3$\mu s$
\item  关断时间$t_{off}$：$\geq 500\mu s$
%\item  恢复时间$t_rr$：1.9$\mu s$
\end{itemize}

对于ABB的晶闸管命名方式，5S代表半导体器件，之后两位字母代表晶闸管类型，如：GF系列门极可关断晶闸管，TP 系列相控晶闸管，GA 系列门极非对称关断晶闸管等，之后两位数字表征额定电流大小，越大额定电流也越大。最后四位数字代表额定电压值如：5STP34H1601、5STP34H1401、5STP34H1201 的额定电压分别为1600V、1400V、1200V。
