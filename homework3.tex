\title{\large{输入驱动电阻和驱动板功率计算 }}
\author{\normalsize{王柯 \ \ \ \ 201613030}}
\date{}
\maketitle

turn on:
\begin{equation}
\begin{split}
  L\frac{\ud i_L}{\ud t}&=V_s  \\ 
  C\frac{\ud u_C}{\ud t}&=-\frac{V_o}{R}  \\
\end{split}
\end{equation}
turn off:
\begin{equation}
\begin{split}
  L\frac{\ud i_L}{\ud t}&=V_s-V_o   \\
  C\frac{\ud u_C}{\ud t}&=i_L-\frac{V_o}{R}  \\
\end{split}
\end{equation}
\begin{equation}
\begin{split}
  \Delta i_L(T)=[DT_sV_s+(1-D)T_s(V_s-V_o)]/L&=0   \\
  \Delta u_C(T)=[\frac{V_o}{R}DT_s+(i_L-\frac{V_o}{R})(1-D)T_s]/C&=0 \\
\end{split}
\end{equation}
therefore
\begin{equation}
\begin{split}
   V_o& =\frac{V_s}{1-D} \\
   I_o& =I_L(1-D)  \\
\end{split}
\end{equation}

$ V_s=12-18\mathrm{V},V_o=36\mathrm{V},D_{min}=\frac{1}{2},D_{max}=\frac{2}{3}$


要使电感电流连续，$ \Delta i_{L}<\frac{2I_{omin}}{1-D}$

$\Delta i_{L}=\frac{V_sDT_s}{L}=\frac{V_oD(1-D)T_s}{L}$

$L>\frac{DV_o(1-D)^2T_s}{2I_{omin}}$，$D(1-D)^2$在$D=\frac{1}{2}$处取得最大值,所以

$L>90\mu H$

不考虑ESR时，

$\Delta v_{pp}^C=\frac{I_oDT_s}{C}\leq2$

$C\geq \frac{I_oDT_s}{2}\cong 33.3\mu F$

考虑ESR时，

忽略输出电流的波动，

$\Delta v_{pp}^R=(I_L+\frac{V_sDT_s}{2L})R_{ESR}$

取$R_{ESR}C\cong 50\mu s,\Delta v_{pp}^R\leq2\mathrm{V}$

可得$C\geq25\times10^{-6}[\frac{V_oDT_s}{2(1-D)L}+\frac{I_o}{1-D}]$

取$D=\frac{2}{3},I_o=5\mathrm{A}$

$C\geq575\mu F$

此时，若考虑输出电流的波动的影响

$v_{pp}^C=\frac{I_oDT_s}{C}\leq0.116\mathrm{V}\ll2\mathrm{V}$，满足要求。
